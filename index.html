        const telegram = window.Telegram.WebApp;


        document.getElementById('agreeBtn').onclick = () => {

            document.getElementById('terms').style.display = 'none';

            document.getElementById('goals').style.display = 'block';

        };


        document.getElementById('disagreeBtn').onclick = () => {

            telegram.close();

        };


        document.querySelectorAll('.goalBtn').forEach(button => {

            button.onclick = () => {

                const selectedGoal = button.getAttribute('data-goal');

                telegram.sendData(`selected_goal:${selectedGoal}`);

                alert(`Вы выбрали цель: ${selectedGoal}`);

            };

        });
